#Base Server install

Enable EPEL

dnf -y install epel-release

# enable CRB repo
/usr/bin/crb enable

Install Ansible core and git

# get my configs
dnf -y install ansible-core git

git clone https://github.com/krispain/rocky.git

cd rocky/ansible

# setup repos etc, one time stuff:
ansible-playbook initial-setup.xml

# Updates
ansible-playbook updates.xml

# install packages etc, will be run over and over
ansible-playbook base.xml

reboot
